int binarySearch(std::vector<int>& arr, int target) {
	int left = 0;
	int right = arr.size() - 1;
				
	while (left <= right) {
		int mid = left + (right - left) / 2;
					
		if (arr[mid] == target)
		return mid;  // Found it!
		else if (arr[mid] < target)
		left = mid + 1;
		else
			right = mid - 1;
		}
return -1;  // Not found
}

			// Outer loop for passes
			for (int i = 0; i < n-1; i++) {
				// Inner loop for comparisons
				for (int j = 0; j < n-i-1; j++) {
					if (arr[j] > arr[j+1]) {
